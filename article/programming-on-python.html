url parsing

https://docs-python.ru/standart-library/modul-urllib-parse-python/funktsija-urlparse-modulja-urllib-parse/

>>> from urllib.parse import urlparse
>>> url = urlparse('https://docs-python.ru:80/search/')
>>> url
# ParseResult(scheme='https', netloc='docs-python.ru:80', path='/search/',
#             params='', query='', fragment='')
>>> url.scheme
# 'https'
>>> url.port
# 80
>>> url.geturl()
# 'http://docs-python.ru:80/search/'
